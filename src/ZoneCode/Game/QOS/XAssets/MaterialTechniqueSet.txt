|| type == MTL_ARG_LITERAL_PIXEL_CONST;

|| type == MTL_ARG_CODE_PIXEL_CONST;

set condition u::nameHash type == MTL_ARG_MATERIAL_VERTEX_CONST
|| type == MTL_ARG_MATERIAL_PIXEL_CONST || type == MTL_ARG_MATERIAL_PIXEL_SAMPLER;
